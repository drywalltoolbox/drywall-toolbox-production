# ============================================================================
# DRYWALL TOOLBOX - Complete Python Dependencies
# ============================================================================
# This file contains all Python package dependencies for the drywall-toolbox
# project, including web scraping, PDF processing, image manipulation, and
# schematic diagram extraction.
#
# Installation: pip install -r requirements.txt
# ============================================================================

# ============================================================================
# WEB SCRAPING & HTTP
# ============================================================================
# Core HTTP library for requests
requests

# Beautiful Soup for HTML/XML parsing
beautifulsoup4

# XML/HTML processing with lxml
lxml

# Cloud scraper to bypass anti-bot protections
cloudscraper

# Mock requests for testing
requests-mock

# ============================================================================
# PDF PROCESSING
# ============================================================================
# PyMuPDF for PDF reading/manipulation
pymupdf

# PyPDF2 for advanced PDF operations (splitting, merging, page extraction)
PyPDF2

# pdf2image for converting PDF pages to images (requires poppler - see instructions)
# pdf2image

# ============================================================================
# IMAGE PROCESSING
# ============================================================================
# Pillow for image manipulation and conversion
Pillow

# ============================================================================
# OCR (Optical Character Recognition)
# ============================================================================
# pytesseract for text extraction from images
pytesseract

# Note: pytesseract requires Tesseract OCR installed on the system:
# Windows: Download installer from https://github.com/UB-Mannheim/tesseract/wiki
# Linux: apt-get install tesseract-ocr
# macOS: brew install tesseract

# ============================================================================
# DATA PROCESSING & VALIDATION
# ============================================================================
# Data validation and serialization
pydantic

# Environment variable management
python-dotenv

# ============================================================================
# SPECIAL INSTALLATION NOTES
# ============================================================================
# 
# PDF to Image Conversion (pdf2image) Requirements:
# 
#   Windows:
#     pip install python-poppler-qt5
#     OR download poppler from: https://github.com/oschwartz10612/poppler-windows/releases/
#   
#   Linux:
#     apt-get install poppler-utils
#   
#   macOS:
#     brew install poppler
#
# OCR Support (pytesseract) Requirements:
#
#   Windows:
#     Download Tesseract installer: https://github.com/UB-Mannheim/tesseract/wiki
#     Run: tesseract-ocr-w64-setup-v5.x.exe (or latest version)
#     After installation, update pytesseract configuration in scripts if needed
#   
#   Linux:
#     apt-get install tesseract-ocr
#   
#   macOS:
#     brew install tesseract
#
# ============================================================================
